    <p>To safely render untrusted URL's you need a combination of
      validation (to remove evil schemes like javascript:) and escaping.</p>
    <p><br>
    </p>
    <p>1) First validate to ensure unsafe schemes and other nonsense is
      rejected. Here is some code to get you started here.</p>
    public static String validateURL(String rawURI)<br>
    throws ValidationException {<br>
    <br>
    // throws URISyntaxException if invalid URI<br>
    URI uri = new URI(rawURI); <br>
    <br>
    // don't allow relative uris<br>
    if (!uri.isAbsolute()) throw new ValidationException("not an
    absolute uri");<br>
    <br>
    // don't allows javascript urls, etc…<br>
    if ((!"http".equals(uri.getScheme()) &&
    (!"https".equals(uri.getScheme())) throw new
    ValidationException("http or https urls are only accepted";<br>
    <br>
    // who legitimately uses user-infos in their urls?!?<br>
    if (uri.getUserInfo() != null)<br>
       throw new ValidationException("this can only be trouble");<br>
    <br>
    // normalize to get rid of '.' and '..' path components<br>
    uri = uri.normalize();<br>
    <br>
    // check: uri.getHost() against whitelist/blacklist?<br>
    // check: uri.getPort() for shenanigans?<br>
    return uri.toASCIIString();<br>
    }<br>
    <br>
    <br>
