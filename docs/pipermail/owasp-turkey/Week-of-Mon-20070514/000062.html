Genel olarak hemen herkes güvenlikte büyük sorumluluğun
yazılımcı/programcıda olduğu konusunda hemfikir. Diğer taraftan belirli
nedenlerden dolayı (zaman, fiyat/performans, ulaşılabilcek verinin
önemsizliği gibi) birçoğumuz güvenlik önlemlerini bildiğimiz halde
projelerimizde uygulamayabiliyoruz. Bu nedenle ben bu tartışmanın birazda en
birincil olarak kodlarımızda uygulayabilceğimiz basit fakat etkili önlemlere
kaymasını öneriyorum. Buna önayak olmak amacıyla genelde en büyük saldırı
noktası olan kullanıcıdan gelen verilerin veritabanında işlenmesi konusunda
benim hemen her form sayfasında kullandığım bir PHP fonksiyonu burada
yazıyorum. Eksiklikleri veya üzerine yapabileceğimiz eklemeleri lütfen
belirtelim. Yalnız şunu da belirteyim bayağı önce yazdığım bir fonksiyon
olduğundan biraz kirli olabilir :)

///////////////
function prepareInputsForSQL($method,$external_array="") {
 $external_array = ($external_array!="") ? $external_array : array("");
 if ($method == "POST") {
  foreach ($_POST as $k=>$v) {
   if (!in_array($k,$external_array)) {
    if (get_magic_quotes_gpc()) {
     $_POST[$k]=stripslashes($v);
    }
    $_POST[$k]=addslashes($v);
   }
  }
 } // if
 else if ($method == "GET") {
  foreach ($_GET as $k=>$v) {
   if (!in_array($k,$external_array)) {
    if (get_magic_quotes_gpc()) {
     $_GET[$k]=stripslashes($v);
    }
    $_GET[$k]=addslashes($v);
   }
  }
 }
}
